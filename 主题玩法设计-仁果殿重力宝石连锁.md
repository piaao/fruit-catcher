## 一、核心机制：重力场 + 宝石连锁

仁果殿是游戏的最终挑战，拥有两套高难度机制，考验玩家的极限操作能力。

---

### 机制 A：重力场（Gravity Field）

#### 设计理念

仁果殿内有强大的重力场，从屏幕中心向外辐射。水果在重力场影响下飞行轨迹弯曲（类似海风但以角色为中心吸引/排斥）。重力场使水果的运动更加不可预测。

#### 视觉效果

- **重力场可视化**：以角色为中心的同心圆，色调为暗金色，半透明旋转光环
- **吸引/排斥效果**：水果靠近/远离时速度相应增减
- **粒子效果**：金色火花粒子沿重力线方向流动
- **背景粒子**：金色火花（`spark`）+ 皇家光芒

#### 机制参数

```
重力模式：吸引（角色引力）或排斥（角色推力），每关随机
重力强度：1级=弱引力, 2级=中引力, 3级=强引力
引力范围：全屏（水果越靠近角色，引力越强）
最大偏移：每帧速度向量偏移 5-20 px/s²（强度决定）
```

#### 实现方案

```javascript
// ---- 重力场系统（仁果殿主题）----
let gravityState = {
  active: false,
  mode: 'attract',    // 'attract'=吸引, 'repel'=排斥
  strength: 0,         // 1=弱, 2=中, 3=强
  fieldAngle: 0,       // 用于旋转动画
};

// 每关初始化
function initGravityLevel() {
  const lvl = LEVELS[level];
  if (!lvl.gravityStrength) { gravityState.active = false; return; }
  gravityState.active = true;
  gravityState.strength = lvl.gravityStrength;
  gravityState.mode = Math.random() < 0.5 ? 'attract' : 'repel';
  playGravitySound(gravityState.mode, gravityState.strength);
}

// 每帧更新水果重力
function updateGravity(dt) {
  if (!gravityState.active) return;

  const baseForce = gravityState.strength === 1 ? 30 : gravityState.strength === 2 ? 70 : 120;
  gravityState.fieldAngle += dt * 0.5; // 旋转速度

  projectiles.forEach(p => {
    if (p.eaten) return;
    const dx = CX - p.x;
    const dy = CY - p.y;
    const d = Math.hypot(dx, dy);
    if (d > 10) {
      const dir = gravityState.mode === 'attract' ? 1 : -1;
      const force = baseForce / Math.max(d / 100, 1); // 距离越近，力越强
      const nx = dx / d;
      const ny = dy / d;
      p.vx += nx * force * dir * dt;
      p.vy += ny * force * dir * dt;
    }
  });
}

// 渲染重力场
function renderGravityField() {
  if (!gravityState.active) return;
  const rings = 5;
  const maxR = Math.min(W, H) * 0.4;
  ctx.save();
  ctx.translate(CX, CY);
  ctx.rotate(gravityState.fieldAngle);
  for (let i = 1; i <= rings; i++) {
    const r = (i / rings) * maxR;
    const alpha = 0.06 - i * 0.008;
    ctx.beginPath();
    ctx.arc(0, 0, r, 0, Math.PI * 2);
    ctx.strokeStyle = `rgba(255,215,0,${alpha})`;
    ctx.lineWidth = 1.5;
    ctx.stroke();
  }
  ctx.restore();
}
```

### 机制 B：宝石连锁（Gem Chain Reaction）

#### 设计理念

某些水果（称为"宝石水果"）被吃掉后会触发连锁反应——在原地生成冲击波，击中附近的水果使其短暂加速并分数翻倍。合理利用连锁可以快速刷分，但宝石水果本身速度较快，需要精准操控。

#### 宝石水果池

| 水果 | 分数 | 速度 | 特殊效果 |
|------|------|------|---------|
| 石榴 💎 | 36分 | 中速 | 吃后产生3层冲击波（各圈半径不同） |
| 榴莲果 🥭 | 40分 | 快 | 吃后产生2层冲击波，范围更大 |

#### 视觉效果

- **宝石状态**：水果周围有金色光圈脉动
- **冲击波**：3层同心圆从中心向外扩散（各层不同颜色/透明度）
- **连锁成功**：被击中的水果变成金色，速度加快
- **连锁失败**：冲击波消散，无效果

#### 实现方案

```javascript
// ---- 宝石连锁系统（仁果殿主题）----
let gemState = {
  active: false,
  chains: [],      // [{x, y, r, maxR, speed, alpha, color, hitFruits: Set}]
  strength: 0,     // 1=石榴, 2=石榴+榴莲
};

function onGemEaten(p) {
  if (!gemState.active) return;
  const isPomegranate = p.def.name === '石榴';
  const isDurian = p.def.name === '榴莲果';
  if (!isPomegranate && !isDurian) return;

  const layers = isPomegranate ? 3 : 2;
  const baseSpeed = isPomegranate ? 150 : 200;
  const colors = isPomegranate
    ? ['rgba(255,215,0,0.4)', 'rgba(255,100,50,0.3)', 'rgba(255,50,100,0.2)']
    : ['rgba(255,215,0,0.5)', 'rgba(255,150,50,0.3)'];
  const maxR = isPomegranate ? 150 : 200;

  for (let i = 0; i < layers; i++) {
    const delay = i * 0.15;
    setTimeout(() => {
      gemState.chains.push({
        x: p.x, y: p.y,
        r: 10,
        maxR: maxR * (1 - i * 0.2),
        speed: baseSpeed - i * 30,
        alpha: 1,
        color: colors[i],
        hitFruits: new Set(),
        layer: i,
      });
      if (typeof playGemChainSound === 'function') playGemChainSound(i);
    }, delay * 1000);
  }
}

function updateGemChains(dt) {
  if (!gemState.active) return;
  gemState.chains.forEach(chain => {
    chain.r += chain.speed * dt;
    chain.alpha = Math.max(0, 1 - chain.r / chain.maxR);

    projectiles.forEach(p => {
      if (p.eaten || p.isBomb || chain.hitFruits.has(p)) return;
      const d = Math.hypot(p.x - chain.x, p.y - chain.y);
      if (d < chain.r && d > chain.r - 20) {
        chain.hitFruits.add(p);
        const cv = Math.hypot(p.vx, p.vy);
        p.vx = p.vx / cv * (cv * 1.5);
        p.vy = p.vy / cv * (cv * 1.5);
        p.gemBoosted = true;
        p.gemBoostTimer = 2.0;
        spawnParticles(p.x, p.y, '#ffd700', 20);
        addFloatingText(p.x, p.y - 30, 'x2', '#ffd700');
      }
    });
  });
  gemState.chains = gemState.chains.filter(c => c.r < c.maxR);
}
```

## 二、BGM 音乐风格

**风格**：皇家庄严，气势恢宏，史诗级

- BGM-1: D小调进行曲，弦乐 + 铜管，皇家入场感
- BGM-2: G大调庆典曲，管弦乐全套，宏大壮丽
- BGM-3: B小调庄严曲，管风琴 + 弦乐，教堂感
- BGM-4: C大调凯旋曲，铜管 + 定音鼓，胜利感
- BGM-5: A小调最终战曲，全套管弦乐 + 合唱效果，史诗级
